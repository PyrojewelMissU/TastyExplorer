export const translateText = async (text, fromLang = 'en', toLang = 'zh-CN') => {
  if (!text || typeof text !== 'string') return text;

  const trimmedText = text.trim();

  // æ£€æŸ¥ç¼“å­˜

    }

    // æ‰€æœ‰å¤±è´¥ï¼Œè¿”å›åŸæ–‡
